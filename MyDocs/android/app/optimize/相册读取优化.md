获取exif 信息来旋转图片角度  
```kotlin
class Adapter{
    var uiScope(Dispatchers.main);
   
    fun onViewHolder{
         // 运行之前
        uiScope.launch{
             // 运行之后
            val job = aync(Dispatchers.IO){
                // 读取开始
                readexif()
                // 读取结束
            }   
            job.await()
            // 读取完毕
        }
    }
}
// 读取文件exif 大概要600ms
```
